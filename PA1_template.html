Are there differences in activity patterns between weekdays and weekends?

daytype <- function(date) {
        if (weekdays(as.Date(date)) %in% c("Saturday", "Sunday")) {
               return("weekend")
        } else {
                return("weekday")
        }
}

data1$daytype <- as.factor(sapply(data1$date, daytype))
